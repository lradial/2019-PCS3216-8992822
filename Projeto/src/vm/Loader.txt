START
	IO 0
	MM POS1
	MM CHECKSUM
	IO 0
	MM POS2
	+ CHECKSUM
	MM CHECKSUM
	IO 0
	MM SIZE
	+ CHECKSUM
	MM CHECKSUM
LOOP
	IO 0
	CN 1
	MM POS1
	+ CHECKSUM
	MM CHECKSUM
	LD POS2
	+ ONE
	MM POS2
	+ CHECKSUM
	MM CHECKSUM
	JZ CARRY
RTN
	LD SIZE
	- ONE
	MM SIZE
	JZ CHECK
	JP LOOP
CARRY
	LD POS1
	+ ONE
	MM POS1
	LD CHECKSUM
	+ ONE
	MM CHECKSUM
	JP RTN
CHECK
	IO 0
	- CHECKSUM
	JZ OK
	OS 1
OK
	OS 0
CHECKSUM K 00
POS1 K 00
POS2 K 00
SIZE K 00
ONE K 01
